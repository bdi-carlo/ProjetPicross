def faireHypothese()

        if @grillesHypothese.empty?
                newGrille = YAML.load(@grillePrim.to_yaml)
                @grillesHypothese.push(newGrille)
        else
                newGrille = YAML.load(@grillesHypothese.last.to_yaml)
                @grillesHypothese.push(newGrille)
        end

        return @grillesHypothese.last
end
