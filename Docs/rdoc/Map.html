    Param : Map à comparer (La classe, pas le tableau)

    Retourne un booléen true si égales, false sinon

    # File Map.rb, line 173
def compare
  0.upto(@rows-1) do |row|
                      0.upto(@cols-1) do |col|
                              if self.accessAt(row,col).verifColor != @res[row][col]
        return FALSE
      end
                      end
              end
  return TRUE
end

def generateSide
  count = 0
  for row in @res do
    lign = Array.new()
    for num in row do
      if num == 1
        count +=1
      end
      if (num == 0 && count != 0 )
        lign.push(count)
        count=0
      end
    end
    if (count != 0 )
      lign.push(count)
      count=0
    end
    @side.push(lign)
  end
  return nil
      end

def generateTop
  count = 0
  0.upto(@cols -1) do |num|
    lign = Array.new()
    for tab in @res do
      if tab[num] == 1
        count +=1
      end
      if (tab[num] == 0 && count != 0 )
        lign.push(count)
        count=0
      end
    end
    if (count != 0 )
      lign.push(count)
      count=0
    end
    @top.push(lign)
  end
end
    import() click to toggle source

    Importe un fichier au format texte en tableau de resultat

    Si le fichier n'existe pas ça plante, il faut mettre le chemin relatif

    # File Map.rb, line 121
def import ()
  fd=File.open(@filename,'r')
  fd.each_line do |x|
    lign=Array.new()
    x.chomp.split(//).each do |char|
      lign.push(char.to_i)
    end
    @res.push(lign)
  end
end
