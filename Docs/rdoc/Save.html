def sauvegarder(unNom, uneGrille)

        @grille=uneGrille
        @nomSave=unNom

        # Serialisation de la grille
        grilleS = @grille.to_yaml()

        # Ecriture de la grille dans le fichier
        monFichier = File.open("../sauvegardes/"+@nomSave, "w")
        monFichier.write(grilleS)
        monFichier.close
end
