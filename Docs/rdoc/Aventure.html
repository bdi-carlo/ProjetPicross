def onDestroy()
  fichier = File.open("../scoreboard/Save_scenar","a")
  puts "QUIIIIIT"
  @window.hide
  fichier.puts("#{@pseudo}-#{@start}")
  fichier.close
  Gtk.main_quit
  @destroy_everything = true
  MenuPrincipal.new(@pseudo)
end
